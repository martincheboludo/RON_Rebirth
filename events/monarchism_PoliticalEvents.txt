add_namespace = monarchism_political

# Monarchism on the Rise
country_event = {
	id = monarchism_political.1
	title = monarchism_political.1.t
	desc = monarchism_political.1.d
	picture = GFX_report_event_fascist_gathering

	trigger = {
		NOT = { has_government = monarchism }
		monarchism > 0.6
		NOT = { monarchism > 0.7 }
	}

	fire_only_once = yes
	
	mean_time_to_happen = {
		days = 30
	}

	# We need a referendum
	option = {
		name = political.14.c
		add_political_power = -100
		set_politics = {
			ruling_party = monarchism
			elections_allowed = no
		}
	}

	# Not right now
	option = {
		name = political.14.e
		ai_chance = {
			factor = 0
		}
		custom_effect_tooltip = impending_civil_war_tt
	}
}

# Neutral Militias Divisions Form
country_event = {
	id = monarchism_political.2
	title = monarchism_political.2.t
	desc = monarchism_political.2.d
	picture = GFX_report_event_fascist_militia
	
	trigger = {
		has_idea_with_trait = centrist_advocate
		NOT = { has_government = monarchism }
		monarchism > 0.3
		NOT = { monarchism > 0.59 }
		NOT = { has_idea = monarchism_militia_divisions }
	}

	mean_time_to_happen = {
		days = 365
		modifier = {
			factor = 0.5
			has_country_flag = monarchism_popular_movement
		}
	}
	
	option = {
		name = monarchism_political.2.a
		add_popularity = {
			ideology = monarchism
			popularity = 0.05
		}
		add_ideas = monarchism_militia_divisions
	}
}

# The Threat of Extremism
country_event = {
	id = monarchism_political.3
	title = monarchism_political.3.t
	desc = monarchism_political.3.d
	picture = GFX_report_event_worker_protests
	
	trigger = {
		has_idea_with_trait = centrist_advocate
		NOT = { has_government = monarchism }
		monarchism > 0.25
		NOT = { fascism > 0.59 }
		OR = {
			any_neighbor_country = {
				has_government = communism
			}
			any_neighbor_country = {
				has_government = fascism
			}
		}
	}

	mean_time_to_happen = {
		days = 365
		modifier = {
			factor = 0.5
			has_country_flag = monarchism_popular_movement
		}
	}
	
	option = {
		name = monarchism_political.3.a
		add_popularity = {
			ideology = monarchism
			popularity = 0.1
		}
	}
}

country_event = {
	id = monarchism_political.5
	title = monarchism_political.5.t
	desc = monarchism_political.5.d
	picture = GFX_report_event_fascist_militia

	trigger = {
		monarchism > 0.7
		num_of_controlled_states > 1
		NOT = { has_government = monarchism }
		has_civil_war = no
	}

	mean_time_to_happen = {
		days = 2
	}

	immediate = {
		hidden_effect = { remove_ideas = monarchism_revolutionaries }
	}

	option = {
		name = monarchism_political.5.a
		ai_chance = {
			factor = 0
		}
		start_civil_war = {
			ideology = monarchism
			size = 0.5
		}
	}

	option = {
		name = monarchism_political.5.b
		ai_chance = {
			factor = 0
		}
		if = {
			limit = {
				has_government = democratic
			}
			start_civil_war = {
				ruling_party = monarchism
				ideology = democratic
				size = 0.5
			}
		}
		if = {
			limit = {
				has_government = communism
			}
			start_civil_war = {
				ruling_party = monarchism
				ideology = communism
				size = 0.5
			}
		}
		if = {
			limit = {
				has_government = fascism
			}
			start_civil_war = {
				ruling_party = monarchism
				ideology = fascism
				size = 0.5
			}
		}
	}

	# We need a referendum
	option = {
		name = political.14.c
		trigger = {
			is_ai = yes
		}
		add_political_power = -100
		set_politics = {
			ruling_party = monarchism
			elections_allowed = no
		}
	}
}
